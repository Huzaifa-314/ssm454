function prime(n){
    var i,j,trace,t=1;
    for(i=3;t <= (n-1);i++){
        for(j=2;j < i;j++){
            if(i%j==0){
                trace=0;
                break;
            }
            else{
                trace=1;
            }
        }
        t=t+trace;
    }
    return i-1;
}

function nprime(x){
    for(var i = 1; prime(i) <= x;i++){
        document.write(prime(i)," ");
    }
}
